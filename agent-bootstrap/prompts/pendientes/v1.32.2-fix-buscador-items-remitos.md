## Descripción

Dos mejoras al formulario de remitos:

1. **Buscador de ítems**: reemplaza el `<select>` simple de insumos por el combobox
   con búsqueda inteligente (mismo patrón de v1.32.0 en facturas de compra).

2. **Selector de empresa**: agrega un campo "Empresa" en los datos del remito,
   pre-seleccionado con la empresa activa del header, pero editable. Permite que el
   usuario elija a qué empresa pertenece el remito antes de guardar.
   El stock es compartido entre empresas — el `company_id` del remito es solo
   para identificación fiscal/contable.

### Función `supplySearch()` para remitos

```javascript
function supplySearch(item, index) {
    return {
        searchText: '',
        results: [],
        showResults: false,
        loading: false,
        highlightedIndex: -1,

        buildLabel(supply) {
            return supply.brand ? `${supply.name} — ${supply.brand}` : supply.name;
        },

        async doSearch() {
            if (this.searchText.length < 2) { this.results = []; this.showResults = false; return; }
            this.loading = true;
            try {
                const resp = await fetch(`{{ route('supplies.search') }}?q=${encodeURIComponent(this.searchText)}`);
                this.results = await resp.json();
                this.showResults = true;
                this.highlightedIndex = -1;
            } catch (e) { this.results = []; } finally { this.loading = false; }
        },

        selectSupply(supply) {
            const label = this.buildLabel(supply);
            item.supply_id     = supply.id;
            item._supply_name  = supply.name;
            item._supply_code  = supply.code;
            item._supply_label = label;
            this.searchText    = '';
            this.showResults   = false;
            this.$nextTick(() => document.querySelector(`#item-qty-${index}`)?.focus());
        },

        unlinkSupply() {
            item.supply_id = ''; item._supply_code = '';
            item._supply_name = ''; item._supply_label = '';
            this.searchText = '';
        },

        highlightNext() { if (this.highlightedIndex < this.results.length - 1) this.highlightedIndex++; },
        highlightPrev() { if (this.highlightedIndex > 0) this.highlightedIndex--; },

        selectHighlighted() {
            if (this.highlightedIndex >= 0) this.selectSupply(this.results[this.highlightedIndex]);
        },

        onTab(event) {
            if (this.showResults && this.highlightedIndex >= 0) { event.preventDefault(); this.selectHighlighted(); }
            else { this.showResults = false; }
        },

        onEnter(event) {
            event.preventDefault();
            if (this.showResults && this.results.length > 0) {
                const idx = this.highlightedIndex >= 0 ? this.highlightedIndex : 0;
                this.selectSupply(this.results[idx]);
            } else {
                this.showResults = false;
                this.$dispatch('add-item');
            }
        },

        init() {
            if (item.supply_id && item._supply_label) { /* badge mode, no tocar searchText */ }
        }
    }
}
```
