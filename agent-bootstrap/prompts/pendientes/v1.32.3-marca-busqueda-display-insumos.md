## Descripción

Mejora el buscador de insumos en facturas de compra (v1.32.2 ya lo aplica a remitos):

1. **Marca en búsqueda y display**: el endpoint incluye `brand` en el criterio de búsqueda
   y en los resultados. El dropdown muestra `Código — Nombre  Marca`.

2. **UX del ítem seleccionado**:
   - Con insumo vinculado: el input desaparece, se muestra **solo badge** `Código  Nombre — Marca  ×`
   - Sin insumo: campo de descripción libre editable
   - Tab: descripción/insumo → Cantidad → Lote → Vencimiento
   - Enter (dropdown cerrado): agrega nueva fila de ítem

## PASO 1 — Actualizar endpoint de búsqueda

En `SupplyController::search()`:

```php
public function search(Request $request)
{
    $q = $request->get('q', '');
    if (strlen($q) < 2) return response()->json([]);

    return response()->json(
        Supply::active()
            ->where(fn($query) => $query
                ->where('name',  'like', "%{$q}%")
                ->orWhere('code',  'like', "%{$q}%")
                ->orWhere('brand', 'like', "%{$q}%"))
            ->orderBy('name')
            ->limit(15)
            ->get(['id', 'code', 'name', 'brand', 'tracks_lot'])
    );
}
```

**Verificación:** `GET /supplies/search?q=labix` retorna insumos con esa marca.

## PASO 2 — Actualizar `supplySearch()` en facturas de compra

Reemplazar la función `supplySearch()` en `purchase-invoices/create.blade.php`
con esta versión que incluye marca, badge-only y Tab/Enter mejorados:

```javascript
function supplySearch(item, index) {
    return {
        searchText: '',
        results: [],
        showResults: false,
        loading: false,
        highlightedIndex: -1,

        buildLabel(supply) {
            return supply.brand ? `${supply.name} — ${supply.brand}` : supply.name;
        },

        async doSearch() {
            if (this.searchText.length < 2) { this.results = []; this.showResults = false; return; }
            this.loading = true;
            try {
                const resp = await fetch(`{{ route('supplies.search') }}?q=${encodeURIComponent(this.searchText)}`);
                this.results = await resp.json();
                this.showResults = true;
                this.highlightedIndex = -1;
            } catch (e) { this.results = []; } finally { this.loading = false; }
        },

        selectSupply(supply) {
            const label = this.buildLabel(supply);
            item.supply_id     = supply.id;
            item.description   = label;
            item._supply_code  = supply.code;
            item._supply_label = label;
            item._tracks_lot   = supply.tracks_lot;
            this.searchText    = '';
            this.showResults   = false;
            // Agregar al array de supplies del invoiceForm para getSupplyTracksLot()
            const parentData = document.querySelector('[x-data^="invoiceForm"]')?.__x?.$data;
            if (parentData?.supplies && !parentData.supplies.find(s => s.id === supply.id)) {
                parentData.supplies.push({ id: supply.id, name: supply.code + ' - ' + supply.name, tracks_lot: supply.tracks_lot });
            }
            this.$nextTick(() => document.querySelector(`#item-qty-${index}`)?.focus());
        },

        unlinkSupply() {
            item.supply_id = ''; item._supply_code = '';
            item._supply_label = ''; item.description = '';
            this.searchText = '';
            this.$nextTick(() => this.$el.querySelector('input[type="text"]')?.focus());
        },

        highlightNext() { if (this.highlightedIndex < this.results.length - 1) this.highlightedIndex++; },
        highlightPrev() { if (this.highlightedIndex > 0) this.highlightedIndex--; },
        selectHighlighted() {
            if (this.highlightedIndex >= 0) this.selectSupply(this.results[this.highlightedIndex]);
        },

        onTab(event) {
            if (this.showResults && this.highlightedIndex >= 0) { event.preventDefault(); this.selectHighlighted(); }
            else {
                this.showResults = false;
                if (!item.supply_id) item.description = this.searchText;
            }
        },

        onEnter(event) {
            event.preventDefault();
            if (this.showResults && this.results.length > 0) {
                const idx = this.highlightedIndex >= 0 ? this.highlightedIndex : 0;
                this.selectSupply(this.results[idx]);
            } else {
                this.showResults = false;
                if (!item.supply_id) item.description = this.searchText;
                this.$dispatch('add-item');
            }
        },

        init() { if (item.description && !item.supply_id) this.searchText = item.description; }
    }
}
```

## PASO 3 — Template de la celda en facturas de compra

Reemplazar el `<td>` de la columna Descripción/Insumo:

```html
<td class="px-3 py-2 relative" x-data="supplySearch(item, index)">
    <input type="hidden" :name="'items[' + index + '][supply_id]'"  :value="item.supply_id || ''">
    <input type="hidden" :name="'items[' + index + '][description]'" :value="item.description || ''">

    {{-- Sin insumo: campo de búsqueda/descripción libre --}}
    <template x-if="!item.supply_id">
        <div>
            <input type="text"
                   x-model="searchText"
                   @input.debounce.300ms="doSearch()"
                   @focus="if (searchText.length >= 2) showResults = true"
                   @keydown.arrow-down.prevent="highlightNext()"
                   @keydown.arrow-up.prevent="highlightPrev()"
                   @keydown.enter="onEnter($event)"
                   @keydown.escape="showResults = false; results = []"
                   @keydown.tab="onTab($event)"
                   placeholder="Buscar insumo o descripción libre..."
                   :id="'item-desc-' + index"
                   class="w-full rounded border-gray-300 text-sm focus:border-zinc-500 focus:ring-zinc-500">

            <div x-show="showResults && results.length > 0" x-cloak
                 @click.outside="showResults = false"
                 class="absolute z-50 left-0 right-0 mt-1 bg-white border border-gray-200 rounded-lg shadow-lg max-h-60 overflow-y-auto">
                <template x-for="(supply, i) in results" :key="supply.id">
                    <button type="button"
                            @click="selectSupply(supply)"
                            @mouseenter="highlightedIndex = i"
                            :class="highlightedIndex === i ? 'bg-teal-50 text-teal-800' : 'text-gray-700'"
                            class="w-full text-left px-3 py-2 text-sm hover:bg-gray-50 flex items-center gap-2 border-b border-gray-100 last:border-0">
                        <span class="font-mono text-xs text-gray-400 shrink-0" x-text="supply.code"></span>
                        <span class="flex-1" x-text="supply.name"></span>
                        <template x-if="supply.brand">
                            <span class="text-xs text-gray-400 shrink-0" x-text="supply.brand"></span>
                        </template>
                    </button>
                </template>
                <div x-show="results.length === 0 && searchText.length >= 2 && !loading"
                     class="px-3 py-2 text-sm text-gray-400">
                    Sin resultados. Escribí descripción libre o presioná Enter para nueva fila.
                </div>
            </div>
        </div>
    </template>

    {{-- Con insumo: solo badge --}}
    <template x-if="item.supply_id">
        <div class="flex items-center gap-2 py-1">
            <span class="font-mono text-xs text-gray-400 shrink-0" x-text="item._supply_code"></span>
            <span class="text-sm text-gray-800 flex-1" x-text="item._supply_label || item.description"></span>
            <button type="button" @click="unlinkSupply()"
                    class="text-gray-300 hover:text-red-500 text-base leading-none shrink-0"
                    title="Desvincular insumo">×</button>
        </div>
    </template>

    {{-- Lote/vencimiento si el insumo controla lote (en las columnas dedicadas) --}}
</td>
```
