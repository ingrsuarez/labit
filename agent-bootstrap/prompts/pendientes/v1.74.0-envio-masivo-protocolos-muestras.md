# v1.74.0 — Envío masivo de protocolos de muestras por email

**Tipo:** minor — nueva funcionalidad
**Prioridad:** alta — requerimiento operativo en producción
**Dependencias:** v1.73.0 (estado `enviado` + columna `sent_at`)

---

## Qué hace

Agrega la capacidad de seleccionar múltiples protocolos en el listado de muestras y
enviarlos por email al mail registrado de cada cliente. Solo se envían los protocolos
validados. Los agrupa por cliente (un email por cliente con todos sus PDFs adjuntos).
Si un cliente no tiene email, permite ingresarlo en el momento o saltear ese cliente.
Al enviar, marca cada protocolo como `enviado` (via `sent_at`).

## PASO 2 — Nueva Mailable para envío múltiple: `SampleBatchMail`

Crear `app/Mail/SampleBatchMail.php`:

```php
<?php

namespace App\Mail;

use App\Models\LabSetting;
use App\Models\Sample;
use Illuminate\Bus\Queueable;
use Illuminate\Database\Eloquent\Collection;
use Illuminate\Mail\Mailable;
use Illuminate\Mail\Mailables\Attachment;
use Illuminate\Mail\Mailables\Content;
use Illuminate\Mail\Mailables\Envelope;
use Illuminate\Queue\SerializesModels;
use PDF;

class SampleBatchMail extends Mailable
{
    use Queueable, SerializesModels;

    public Collection $samples;

    public ?string $customMessage;

    public ?string $signature;

    public function __construct(Collection $samples, ?string $customMessage = null)
    {
        $this->samples = $samples;
        $this->customMessage = $customMessage;
        $this->signature = LabSetting::get('results_signature', '');
    }

    public function envelope(): Envelope
    {
        $count = $this->samples->count();
        $subject = $count === 1
            ? LabSetting::get('results_default_subject', 'Informe de Resultados')
            : "Informes de Resultados ({$count} protocolos)";

        if ($count === 1) {
            $subject = str_replace('{protocol}', $this->samples->first()->protocol_number, $subject);
        }

        return new Envelope(subject: $subject);
    }

    public function content(): Content
    {
        return new Content(view: 'emails.sample-batch-result');
    }

    public function attachments(): array
    {
        $attachments = [];

        foreach ($this->samples as $sample) {
            $sample->load([
                'customer',
                'determinations.test.parentTest',
                'determinations.test.parentTests',
                'determinations.test.children',
                'determinations.test.childTests',
                'determinations.test.defaultReferenceCategory',
                'determinations.referenceCategory',
                'determinations.determinationValidator',
                'creator',
                'validator',
            ]);

            $pdf = PDF::loadView('sample.pdf-mpdf', ['sample' => $sample], [], [
                'margin_top' => 35,
                'margin_bottom' => 20,
                'margin_left' => 15,
                'margin_right' => 15,
            ]);

            $filename = SampleResultMail::generatePdfFilenamePublic($sample);

            $attachments[] = Attachment::fromData(
                fn () => $pdf->output(),
                $filename
            )->withMime('application/pdf');
        }

        return $attachments;
    }
}
```

**Nota:** `SampleResultMail::generatePdfFilename()` es privado. Hacer el método `protected static`
(cambiar `private` por `protected`) en `SampleResultMail` y llamarlo desde `SampleBatchMail`
como `SampleResultMail::generatePdfFilename($sample)`. Alternativamente, duplicar la lógica
del nombre de archivo en `SampleBatchMail`.

## PASO 4 — Implementar `SampleController::batchEmail()`

```php
public function batchEmail(Request $request)
{
    $this->authorize('samples-reports.send');

    $request->validate([
        'sample_ids'        => 'required|array|min:1',
        'sample_ids.*'      => 'integer|exists:samples,id',
        'email_overrides'   => 'nullable|array',
        'email_overrides.*' => 'nullable|email',
    ]);

    $fromEmail = LabSetting::get('results_email', config('mail.from.address'));
    $fromName  = LabSetting::get('results_from_name', config('mail.from.name'));

    $samples = Sample::with(['customer', 'determinations'])
        ->whereIn('id', $request->sample_ids)
        ->get();

    // Filtrar solo los validados
    $validated = $samples->filter(fn (Sample $s) => $s->isValidated());

    // Agrupar por customer_id
    $grouped = $validated->groupBy('customer_id');

    $results = [
        'sent'    => [],
        'skipped' => [],
        'errors'  => [],
    ];

    // Protocolos no validados → skipped
    $notValidated = $samples->filter(fn (Sample $s) => ! $s->isValidated());
    foreach ($notValidated as $s) {
        $results['skipped'][] = $s->protocol_number.' (no validado)';
    }

    foreach ($grouped as $customerId => $customerSamples) {
        $customer = $customerSamples->first()->customer;

        // Determinar email destino: override tiene prioridad, luego email del cliente
        $emailOverrides = $request->input('email_overrides', []);
        $toEmail = $emailOverrides[$customerId] ?? $customer?->email ?? null;

        if (! $toEmail) {
            foreach ($customerSamples as $s) {
                $results['skipped'][] = $s->protocol_number.' (sin email del cliente)';
            }
            continue;
        }

        try {
            Mail::mailer('smtp')
                ->to($toEmail)
                ->send(
                    (new \App\Mail\SampleBatchMail($customerSamples))
                        ->from($fromEmail, $fromName)
                );

            // Marcar cada protocolo como enviado
            foreach ($customerSamples as $s) {
                $s->update(['sent_at' => now()]);
                $s->logAudit('email_sent', "Enviado en lote a {$toEmail}");
                $results['sent'][] = $s->protocol_number;
            }
        } catch (\Exception $e) {
            foreach ($customerSamples as $s) {
                $results['errors'][] = $s->protocol_number.' (error: '.$e->getMessage().')';
            }
        }
    }

    return response()->json($results);
}
```

## PASO 6 — Actualizar Alpine.js `sampleFilter()` en `index.blade.php`

Agregar las siguientes propiedades y métodos al objeto Alpine:

```javascript
// Propiedades nuevas
selectedIds: [],
selectAll: false,
showBatchModal: false,
batchData: { groups: [], skipped: [] },
batchSending: false,
batchResult: null,

// Datos de cada sample para el modal (serializar desde Blade)
sampleMeta: @json($samples->map(fn($s) => [
    'id' => $s->id,
    'protocol_number' => $s->protocol_number,
    'customer_id' => $s->customer_id,
    'customer_name' => $s->customer?->name ?? 'N/A',
    'customer_email' => $s->customer?->email ?? null,
    'is_validated' => $s->isValidated(),
])->values()),

// Métodos nuevos
toggleAll() {
    if (this.selectAll) {
        // Seleccionar solo los validados visibles
        this.selectedIds = this.sampleMeta
            .filter(s => s.is_validated)
            .map(s => s.id);
    } else {
        this.selectedIds = [];
    }
},

openBatchModal() {
    const selected = this.sampleMeta.filter(s => this.selectedIds.includes(s.id));
    const validated = selected.filter(s => s.is_validated);
    const skipped = selected.filter(s => !s.is_validated);

    // Agrupar validados por cliente
    const groups = {};
    for (const s of validated) {
        if (!groups[s.customer_id]) {
            groups[s.customer_id] = {
                customer_id: s.customer_id,
                customer_name: s.customer_name,
                email: s.customer_email || '',
                has_email: !!s.customer_email,
                skip: false,
                samples: [],
            };
        }
        groups[s.customer_id].samples.push(s);
    }

    this.batchData = {
        groups: Object.values(groups),
        skipped: skipped,
    };
    this.showBatchModal = true;
},

async sendBatch() {
    this.batchSending = true;
    const emailOverrides = {};
    const sampleIds = [];

    for (const group of this.batchData.groups) {
        if (group.skip) continue;
        for (const s of group.samples) {
            sampleIds.push(s.id);
        }
        emailOverrides[group.customer_id] = group.email;
    }

    try {
        const res = await fetch('{{ route("sample.batch-email") }}', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
                'X-CSRF-TOKEN': document.querySelector('meta[name="csrf-token"]').content,
            },
            body: JSON.stringify({
                sample_ids: sampleIds,
                email_overrides: emailOverrides,
            }),
        });
        this.batchResult = await res.json();
    } catch (e) {
        this.batchResult = { sent: [], skipped: [], errors: ['Error de red'] };
    }

    this.batchSending = false;
    this.showBatchModal = false;
    this.selectedIds = [];
    this.selectAll = false;
},
```
