```bash

git push origin develop

# Merge a master
git checkout master
git pull origin master --ff-only

# Tag
git tag -a v1.0.0 -m "Release v1.0.0: Línea base del proyecto"

# Limpieza
git branch -d feature/v1.0.0-linea-base
```
